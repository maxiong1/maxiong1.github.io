# 椭圆角度倾斜、计算椭圆边缘的xy坐标

    biaopanhudu(per, type) {
        // 计算上面左右两侧弧度
        // right 527,150(x,y)  40,120
        // 0 , -13
        // 20, -19.3

        // 21, -21.5
        // 30 - 21.5 - 7.8/2 = 3.9
        // 40, -29.3

        // 41, -31
        // 60, -39.6

        // 61, -42.5
        // 80, -51.5

        // 81, -54.5
        //100, -70
        if (type === 'right') {
            let radianNow = 0
            if (per >= 0 && per <= 20) {
                radianNow = -13 - (6.3 * per) / 20
            } else if (per > 20 && per <= 40) {
                radianNow = -21.5 - ((per - 20) / 20) * 7.8
            } else if (per > 40 && per <= 60) {
                radianNow = -31 - ((per - 40) / 20) * 8.6
            } else if (per > 60 && per <= 80) {
                radianNow = -42.5 - ((per - 60) / 20) * 9
            } else if (per > 80 && per <= 100) {
                radianNow = -54.5 - ((per - 80) / 20) * 15.5
            }
            let theta = (radianNow * Math.PI) / 180
            // 旋转之前的xy
            let x1 = 40 * Math.cos(theta)
            let y1 = 120 * Math.sin(theta)
            if(document.querySelector('.biao007_02')){
                document.querySelector('.biao007_02').style.left = x1 + 527 + 3 + 'px'
                document.querySelector('.biao007_02').style.top = y1 + 150 - 5 + 'px'
            }
        } else if (type === 'left') {
            // 0 -167 6.5
            // 20 -160.5

            // 21, -158.5
            // 40, -150.5

            // 41, -148.5 8.7
            // 60, -139.8

            // 61, -137.8 5
            // 80, -128.8

            // 81, -124.9 14
            //100, -110.9
            let radianNow = 0
            if (per >= 0 && per <= 20) {
                radianNow = -167 + (6.5 * per) / 20
            } else if (per > 20 && per <= 40) {
                radianNow = -158.5 + ((per - 20) / 20) * 8
            } else if (per > 40 && per <= 60) {
                radianNow = -148.5 + ((per - 40) / 20) * 8.7
            } else if (per > 60 && per <= 80) {
                radianNow = -137.8 + ((per - 60) / 20) * 9
            } else if (per > 80 && per <= 100) {
                radianNow = -124.9 + ((per - 80) / 20) * 14
            }
            let theta = (radianNow * Math.PI) / 180
            // 旋转之前的xy
            let x1 = 40 * Math.cos(theta)
            let y1 = 120 * Math.sin(theta)
            if(document.querySelector('.biao007_01')){
                document.querySelector('.biao007_01').style.left = x1 + 133 - 13 + 'px'
                document.querySelector('.biao007_01').style.top = y1 + 150 - 5 + 'px'
            }
        }
    },
