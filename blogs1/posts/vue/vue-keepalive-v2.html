vue2 keepalive 测试文章 ​

需要对路由进行扁平化处理，不然keepalive对多层routerView获取到的组件name值是routeView

路由缓存 ​

vnode.data.keepAlive = true是起到组件缓存重要指标;

有include与exclude属性 ​

  • 如果是多级路由的情况下，缓存会失败：原因
js
// 在keepalive源码render函数中
const slot = this.$slots.default
// vnode 是没有被包含在具名插槽中的节点（数组），for循环，当前节点具有 componentOptions 属性 或者 isComment && asyncFactory 都为true;
// 可知如果是多级路由，那么获取的vnode是顶级路由的组件
const vnode = getFirstComponentChild(slot)
const componentOptions = vnode && vnode.componentOptions
if (componentOptions) {
	// 从componentOptions.Ctor.options 或者 componentOptions.tag 拿取name属性值
	// 此时的name值为顶级路由的name值
	const name = _getComponentName(componentOptions)
	const { include, exclude } = this
	if (
		// not included--不在include
		// include有值 && （name不存在或者include中不包含name）
		(include && (!name || !matches(include, name))) ||
		// excluded--在exclude
		// exclude有值 && name && 包含name
		(exclude && name && matches(exclude, name))
	) {
		// 在这里就结束了，该name组件不具备缓存
		return vnode
	}
	//...
}
  • 如果是多级路由（有处理）的情况下，缓存会正常：
js
// 处理方法 matched：当前路由的所有嵌套路径片段的路由记录
// 这样会使页面组件暴露在最外层
router.beforeEach((to, from, next) => {
	// 多级菜单会导致嵌套dom多
	console.log(to.matched)
	if (to.matched && to.matched.length >1) {
		to.matched.splice(0, to.matched.length - 1)
	}
	//console.log(to.matched)
	next()
})

这样处理以后他会绕过render函数内的include exclude条件判断，会走到vnode.data.keepAlive = true

  • 将顶级路由组件name，添加到include也会生效，这样会出现一些问题,所有子路由都缓存了

  • 可能还会存在其他方案

如果出现（同一组件）动态路由的时候,缓存异常（相同） ​

  • 首先我们对router-view添加一个key="$route.fullpath"
    • 出现问题：就会出现我关闭当前页再打开的时候上次的缓存数据还在，正常逻辑应该是缓存数据清空
    • 解决方法
      • 在当前组件内清除缓存
        • 首先需要对多级路由进行处理，让页面组件暴露在keep-alive之下。如果不这样在当前组件使用this.$vnode发现获取的parent属性值是undefined，说明router-view是函数式组件
        • 使用this.$vnode.parent获取到keep-alive组件的componentInstance
        • 再配合watch对keepalive的(cache、keys)属性进行清除,如果使用beforeRouteLeave会出现在关闭非当前页面的时候缓存无法清除的情况
      • 修改keepalive源码
        • 添加对key支持
        • 添加includeKey excludeKey属性来替换include exclude
        • 修改源码的方案虽然简单，但是会存在问题，每创建一个项目修改一次源码这样不好，所以我们要封装一个组件来替换keep-alive

function pruneCacheByKey(keepAliveInstance, filter) {
 var cache = keepAliveInstance.cache, keys = keepAliveInstance.keys, _vnode = keepAliveInstance._vnode;
 for (var key in cache) {
     var entry = cache[key];
     if (entry) {

		    cache[key] = null;
		    remove$2(keys, key);
		 }
     }
 }

export default {
  name: 'keep-alive',
  abstract: true,

  props: {
    include: patternTypes,
    exclude: patternTypes,
    max: [String, Number]
  },

  methods: {
    cacheVNode() {
      const { cache, keys, vnodeToCache, keyToCache } = this
      if (vnodeToCache) {
        const { tag, componentInstance, componentOptions } = vnodeToCache
        cache[keyToCache] = {
          name: _getComponentName(componentOptions),
          tag,
          componentInstance
        }
        keys.push(keyToCache)
        // prune oldest entry
        if (this.max && keys.length > parseInt(this.max)) {
          pruneCacheEntry(cache, keys[0], keys, this._vnode)
        }
        this.vnodeToCache = null
      }
    }
  },

  created() {
    this.cache = Object.create(null)
    this.keys = []
  },

  destroyed() {
    for (const key in this.cache) {
      pruneCacheEntry(this.cache, key, this.keys)
    }
  },

  mounted() {
    this.cacheVNode()
    this.$watch('include', val => {
      pruneCache(this, name => matches(val, name))
    })
    this.$watch('exclude', val => {
      pruneCache(this, name => !matches(val, name))
    })
	// start 新增 includeKey  excludeKey
	this.$watch('include', val => {
	  pruneCacheByKey(this, name => typeof name === 'string' && matches(val, name))
	})
	this.$watch('exclude', val => {
	  pruneCacheByKey(this, name => typeof name !== 'string' || !matches(val, name))
	})
	// end 新增 includeKey excludeKey
  },

  updated() {
    this.cacheVNode()
  },

  render() {
    const slot = this.$slots.default
    const vnode = getFirstComponentChild(slot)
    const componentOptions = vnode && vnode.componentOptions
    if (componentOptions) {
      // check pattern
      const name = _getComponentName(componentOptions)
      const { include, exclude } = this
	  // 将下面的key拿上来
	  const key =
	    vnode.key == null
	      ? // same constructor may get registered as different local components
	        // so cid alone is not enough (#3269)
	        componentOptions.Ctor.cid +
	        (componentOptions.tag ? `::${componentOptions.tag}` : '')
	      : vnode.key
      if (
        // not included
        (include && (!name || !matches(include, name))) ||
        // excluded
        (exclude && name && matches(exclude, name)) ||
		// start 新增 includeKey excludeKey
			(includeKey && (typeof key !== 'string' || !matches(includeKey, key))) ||
		    (excludeKey && typeof key === 'string' && matches(excludeKey, key))
		// end 新增 includeKey excludeKey
      ) {
        return vnode
      }

      const { cache, keys } = this
     // 这里的定义的key拿到最上面去
      if (cache[key]) {
        vnode.componentInstance = cache[key].componentInstance
        // make current key freshest
        remove(keys, key)
        keys.push(key)
      } else {
        // delay setting the cache until update
        this.vnodeToCache = vnode
        this.keyToCache = key
      }

      // @ts-expect-error can vnode.data can be undefined
      vnode.data.keepAlive = true
    }
    return vnode || (slot && slot[0])
  }
}

function pruneCache(keepAliveInstance, filter) {
	var cache = keepAliveInstance.cache,
		keys = keepAliveInstance.keys,
		_vnode = keepAliveInstance._vnode;
	for (var key in cache) {
		var entry = cache[key];
		if (entry) {
			var name_1 = entry.name;

			}
		}
	}

function pruneCacheEntry(cache, key, keys, current) {
	var entry = cache[key];
	if (entry && (!current || entry.tag !== current.tag)) {
		// @ts-expect-error can be undefined
		entry.componentInstance.$destroy();
	}
	cache[key] = null;
	remove$2(keys, key);
}
